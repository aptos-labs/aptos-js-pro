# useBalance

Hook for getting the current fungible asset or coin balance of an account.

## Usage

```tsx
import { useBalance } from "@aptos-labs/react";

export default function App() {
  const { data: aptBalance, isLoading } = useBalance({
    address:
      "0x926ff619e85bdb5b02680f27bea5ea3302c71f86be4b16fa0d63df213add927a",
    asset: "0xa",
  });

  if (isLoading) {
    return <div>Loading...</div>;
  }

  return <div>APT Balance: {aptBalance} octas</div>;
}
```
